##########################################
### Build dsniff
##########################################
# Debian 
sudo apt-get install gcc autoconf automake m4 pkg-config
sudo apt-get install libdb-dev libpcap0.8-dev libnids-dev libssl-dev libxmu-dev libnet1-dev libtirpc-dev

# Fedora
sudo dnf install gcc autoconf automake m4 pkg-config
sudo dnf install libdb-devel libpcap-devel libnids-devel openssl-devel libXmu-devel libnet-devel libtirpc-devel rpcgen libnsl2-devel

# Arch Linux
sudo pacman -S gcc autoconf automake m4 pkg-config
sudo pacman -S rpcsvc-proto glibc libpcap openssl libxmu glib2 libnet libnids libtirpc


wget https://src.fedoraproject.org/repo/pkgs/dsniff/dsniff-2.4b1.tar.gz/2f761fa3475682a7512b0b43568ee7d6/dsniff-2.4b1.tar.gz
tar zxvf dsniff-2.4b1.tar.gz

# Debian r29 & Arch Bug Patch
wget https://raw.githubusercontent.com/Mint-Fans/linux-package/patch/dsniff-2.4-r29.patch
patch -p0 -i dsniff-2.4-r29.patch

# fprintf Patch
sed -i 's/fprintf(stderr, /printf(/g' dsniff-2.4/arpspoof.c

# Build
cd dsniff-2.4
autoreconf -fiv
./configure --prefix=/usr --with-libtirpc
make -j$(nproc)
make install_prefix=$HOME/Build/dsniff install
sudo make install

or
make arpspoof
